KSambaRecord& KSambaRecord::operator=(const KSambaRecord &aRec)
{
  //assignment operator

  if(&aRec == this) return *this;

  this->KSubRecord::operator=(aRec);
  CopyLocalMembers(aRec);

  return *this;
}
